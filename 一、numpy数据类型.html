一、numpy数据类型

#基本数据类型
print(np.int8(-1))
-1
print(np.uint8(-1))
255
print(np.int16(-1))
-1
print(np.uint16(-1))
65535
print(np.int32(-1))
-1
print(np.uint32(-1))
4294967295
print(np.int64(-1))
-1
print(np.uint64(-1))
18446744073709551615
print(np.float16(42))
42.0
print(np.float32(42))
42.0
print(np.float64(42))
42.0
print(np.float64(True))
1.0
print(np.complex64(12))
(12+0j)
print(np.complex128(12))
(12+0j)
print(np.bool(1000))
True
print(np.unicode(10))
10

#指定数组的数据类型
m = np.array(7,dtype = np.uint16)
print(m)
7
m = np.array(7,dtype = np.float64)
print(m)
7.0

#数据类型转换
arr = np.array([1,2,3,4,5])
print(arr.dtype)#获取numpy数组的数据类型
int32
print(arr)
[1 2 3 4 5]
float_arr = arr.astype(np.float64)#将np数组转换成float64
print(float_arr.dtype)#获取numpy数组的数据类型
float64
print(float_arr)
[ 1.  2.  3.  4.  5.]
print(float_arr.dtype.itemsize)#元素字节数
8

#字符编码
print(np.arange(7,dtype = 'f'))#f代表float32
[ 0.  1.  2.  3.  4.  5.  6.]
print(np.dtype('f'))##获取numpy数组的数据类型
float32
print(np.arange(7,dtype = 'D'))#D代表复数
[ 0.+0.j  1.+0.j  2.+0.j  3.+0.j  4.+0.j  5.+0.j  6.+0.j]
print(np.dtype('D'))#获取numpy数组的数据类型
complex128

#dtype类的属性
t = np.dtype('float64')
print(t.char)#获得float64的字符编码
d
print(t.type)#获得float64的numpy类型
<class 'numpy.float64'>
print(t.str)#字符编码
<f8

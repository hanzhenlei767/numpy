#将条件逻辑表达为数组运算
xarr = np.array([1.1, 1.2, 1.3, 1.4, 1.5])
yarr = np.array([2.1, 2.2, 2.3, 2.4, 2.5])
cond = np.array([True, False, True, True, False])
result = [(x if c else y)
          for x, y, c in zip(xarr, yarr, cond)]#根据cond选择是xarr或yarr中元素
print(result)
[1.1000000000000001, 2.2000000000000002, 1.3, 1.3999999999999999, 2.5]
#等价于上面的代码
result = np.where(cond, xarr, yarr)#cond向量维度和后面的一致，cond真返回xarr元素，假返回yarr元素。向量的三目运算符
print(result)
[ 1.1  2.2  1.3  1.4  2.5]

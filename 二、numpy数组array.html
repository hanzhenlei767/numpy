二、numpy数组array

a = np.arange(5)
print(type(a))#数组类型
<class 'numpy.ndarray'>
print(a)
[0 1 2 3 4]
print(a.dtype)#获取numpy数组的数据类型
int32
print(a.shape)#维度
(5,)



#numpy利用它的array创建多维数组

data=[[1,1,1],[1,1,1],[1,1,1]]#list类型
m = np.array(data)#将list类型转换成numpy.ndarray类型
m = np.asarray(data)#与上一行等价
print(m)
[[1 1 1]
[1 1 1]
[1 1 1]]

m = np.array(2)#将2转换成np的array类型
print(m)
2
m = np.array([np.array(2),np.array(2)])#将list转换成np的array类型
print(m)
[2 2]

#numpy利用它的zeros创建数组
m = np.zeros(5)#一维数组，5个0
print(m)
[ 0.  0.  0.  0.  0.]
m = np.zeros((3,2))#3行2列数组，全是0
print(m)
[[ 0.  0.]
[ 0.  0.]
[ 0.  0.]]

#numpy利用它的ones创建数组
m = np.ones(4)#全是1的数组
print(m)
[ 1.  1.  1.  1.]
m = np.ones((3,2))
print(m)
[[ 1.  1.]
[ 1.  1.]
[ 1.  1.]]

#numpy利用它的eye创建单位矩阵
m = np.eye(3,3)#单位矩阵
print(m)
[[ 1.  0.  0.]
[ 0.  1.  0.]
[ 0.  0.  1.]]
